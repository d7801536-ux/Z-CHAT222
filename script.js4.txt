// UTILS
function showToast(msg) {
    const container = document.getElementById('toast-container');
    const toast = document.createElement('div');
    toast.className = 'toast';
    toast.innerText = msg;
    container.appendChild(toast);
    setTimeout(() => toast.remove(), 3000);
}

function loadTheme() {
    const theme = localStorage.getItem('zchat_theme') || 'theme-neon';
    document.body.className = theme;
}

// ================= AUTH SYSTEM =================
toggleAuthMode.addEventListener('click', () => {
    isSignUpMode = !isSignUpMode;
    authUsername.classList.toggle('hidden', !isSignUpMode);
    authUsername.required = isSignUpMode;
    toggleAuthMode.innerText = isSignUpMode ? "Already have an account? Sign In" : "Need an account? Sign Up";
    authModeText.innerText = isSignUpMode ? "Create Identity Node" : "Access Secure Node";
    authSubmitBtn.innerText = isSignUpMode ? "Register" : "Initialize Uplink";
});

authForm.addEventListener('submit', async (e) => {
    e.preventDefault();
    const email = authEmail.value;
    const pass = authPassword.value;
    
    try {
        if (isSignUpMode) {
            const username = authUsername.value.trim();
            const userCred = await createUserWithEmailAndPassword(auth, email, pass);
            await set(ref(db, `users/${userCred.user.uid}`), {
                uid: userCred.user.uid,
                username: username,
                email: email,
                profilePic: 'https://via.placeholder.com/150',
                status: 'Available',
                createdAt: serverTimestamp()
            });
            showToast("Account created!");
        } else {
            await signInWithEmailAndPassword(auth, email, pass);
            showToast("Login successful!");
        }
    } catch (err) {
        showToast("Auth Error: " + err.message);
    }
});

document.getElementById('google-login-btn').addEventListener('click', async () => {
    try {
        const provider = new GoogleAuthProvider();
        const res = await signInWithPopup(auth, provider);
        const userRef = ref(db, `users/${res.user.uid}`);
        const snap = await get(userRef);
        if(!snap.exists()) {
            await set(userRef, {
                uid: res.user.uid,
                username: res.user.displayName || 'User_' + Date.now().toString().slice(-4),
                email: res.user.email,
                profilePic: res.user.photoURL || 'https://via.placeholder.com/150',
                status: 'Available'
            });
        }
    } catch (err) { showToast("Google Auth Failed"); }
});

onAuthStateChanged(auth, async (user) => {
    if (user) {
        currentUser = user;
        onValue(ref(db, `users/${user.uid}`), (snap) => {
            userData = snap.val();
            document.getElementById('settings-username').value = userData.username;
            document.getElementById('settings-status').value = userData.status;
        });
        authScreen.classList.add('hidden');
        appScreen.classList.remove('hidden');
        loadSidebar();
    } else {
        currentUser = null;
        userData = null;
        authScreen.classList.remove('hidden');
        appScreen.classList.add('hidden');
    }
});

// ================= SIDEBAR & USERS =================
function loadSidebar() {
    onValue(ref(db, 'users'), (snapshot) => {
        sidebarList.innerHTML = '';
        snapshot.forEach(child => {
            const user = child.val();
            if(user.uid === currentUser.uid) return; // Skip self
            
            const div = document.createElement('div');
            div.className = 'chat-item';
            div.innerHTML = `
                <img src="${user.profilePic}" class="chat-avatar">
                <div class="chat-info">
                    <div class="chat-name">${user.username}</div>
                    <div class="chat-preview">${user.status}</div>
                </div>
            `;
            div.onclick = () => openChat(user);
            sidebarList.appendChild(div);
        });
    });
}

// ================= CHAT & MESSAGES =================
function openChat(targetUser) {
    currentChatId = [currentUser.uid, targetUser.uid].sort().join('_');
    document.getElementById('header-name').innerText = targetUser.username;
    document.getElementById('header-avatar').src = targetUser.profilePic;
    document.getElementById('header-status-text').innerText = targetUser.status;
    document.getElementById('call-btn').classList.remove('hidden');
    
    // Attach event for Call UI
    document.getElementById('call-btn').onclick = () => startCall(targetUser);
    
    onValue(ref(db, `messages/${currentChatId}`), (snapshot) => {
        messagesArea.innerHTML = '';
        snapshot.forEach(child => {
            const msg = child.val();
            const isSent = msg.senderId === currentUser.uid;
            const div = document.createElement('div');
            div.className = `message ${isSent ? 'sent' : 'received'}`;
            
            let content = '';
            if(msg.type === 'image') {
                content = `<img src="${msg.imageUrl}" class="message-img" onclick="window.open('${msg.imageUrl}')">`;
            }
            if(msg.text) {
                content += `<div class="message-bubble">${msg.text}</div>`;
            }
            
            div.innerHTML = `
                <div class="msg-sender">${isSent ? 'You' : targetUser.username}</div>
                ${content}
                <div class="message-time">${new Date(msg.timestamp).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}</div>
            `;
            
            // Allow selecting message for Z-Scanner
            div.querySelector('.message-bubble')?.addEventListener('dblclick', () => {
                activeScanMessage = msg.text;
                document.getElementById('scan-target-msg').innerText = `"${msg.text}"`;
                document.getElementById('scanner-modal').classList.remove('hidden');
            });
            
            messagesArea.appendChild(div);
        });
        messagesArea.scrollTop = messagesArea.scrollHeight;
    });
}

// ================= SENDING & IMAGES =================
attachBtn.onclick = () => imageUploadInput.click();
imageUploadInput.onchange = (e) => {
    if(e.target.files[0]) {
        selectedImageFile = e.target.files[0];
        imagePreviewImg.src = URL.createObjectURL(selectedImageFile);
        imagePreviewContainer.classList.remove('hidden');
    }
};
cancelImageBtn.onclick = () => {
    selectedImageFile = null;
    imagePreviewContainer.classList.add('hidden');
    imageUploadInput.value = '';
};

sendBtn.onclick = async () => {
    if(!currentChatId) return showToast("Select a chat first!");
    const text = messageInput.value.trim();
    if(!text && !selectedImageFile) return;
    
    sendBtn.disabled = true;
    const msgData = {
        senderId: currentUser.uid,
        timestamp: Date.now()
    };
    
    try {
        if(selectedImageFile) {
            const fileRef = storageRef(storage, `chat_images/${Date.now()}_${selectedImageFile.name}`);
            await uploadBytes(fileRef, selectedImageFile);
            msgData.imageUrl = await getDownloadURL(fileRef);
            msgData.type = 'image';
        } else {
            msgData.type = 'text';
        }
        if(text) msgData.text = text;
        
        await push(ref(db, `messages/${currentChatId}`), msgData);
        
        messageInput.value = '';
        cancelImageBtn.click();
    } catch(err) {
        showToast("Failed to send: " + err.message);
    }
    sendBtn.disabled = false;
};

        showToast("Profile Updated!");
        document.getElementById('settings-modal').classList.add('hidden');
    } catch(err) {
        showToast("Update Failed");
    }
};

// ================= CALL SYSTEM UI =================
function startCall(user) {
    const overlay = document.getElementById('call-overlay');
    document.getElementById('call-name').innerText = `Calling ${user.username}...`;
    document.getElementById('call-avatar').src = user.profilePic;
    overlay.classList.remove('hidden');
    
    let seconds = 0;
    callInterval = setInterval(() => {
        seconds++;
        const mins = String(Math.floor(seconds / 60)).padStart(2, '0');
        const secs = String(seconds % 60).padStart(2, '0');
        document.getElementById('call-timer').innerText = `${mins}:${secs}`;
    }, 1000);
}

document.getElementById('end-call-btn').onclick = () => {
    document.getElementById('call-overlay').classList.add('hidden');
    clearInterval(callInterval);
    document.getElementById('call-timer').innerText = "00:00";
    showToast("Call Ended");
};

document.getElementById('z-scanner-btn').onclick = () => {
    document.getElementById('scan-target-msg').innerText = "Double-click any chat bubble to select a message for scanning.";
    document.getElementById('scan-results').innerHTML = "";

document.getElementById('run-scan-btn').onclick = async () => {
    if(!activeScanMessage) return showToast("Select a message first!");
    
    const resultsDiv = document.getElementById('scan-results');
    resultsDiv.innerHTML = "<div class='dot-animate'></div> Scanning global database...";
    
    try {
        // Ping Wikipedia API to Fact Check
        const url = `https://en.wikipedia.org/w/api.php?action=query&list=search&srsearch=${encodeURIComponent(activeScanMessage)}&utf8=&format=json&origin=*`;
        const res = await fetch(url);
        const data = await res.json();
        
        if(data.query.search.length > 0) {
            const topResult = data.query.search[0];
            resultsDiv.innerHTML = `
                <div style="color: #4ade80; font-weight: bold; margin-bottom: 5px;">[VERIFIED DATAPOINT FOUND]</div>
                <strong style="color:var(--primary)">${topResult.title}</strong><br>
                <div style="margin-top: 5px; font-size: 0.9em;">${topResult.snippet}...</div>
            `;
        } else {
            resultsDiv.innerHTML = `<div style="color: #ff4444; font-weight: bold;">[NO MATCH]</div><p>Information could not be verified in the public databanks.</p>`;
        }
    } catch(err) {
        resultsDiv.innerHTML = "API Connection Failed.";
    }
};
